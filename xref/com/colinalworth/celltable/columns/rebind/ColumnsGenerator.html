49  public class ColumnsGenerator extends Generator {
50  	//private TreeLogger logger;
51  	private GeneratorContext context;
52  
53  	private Set<String> names = new HashSet<String>();
54  
55  	@Override
56  	public String generate(TreeLogger logger, GeneratorContext context, String typeName) throws UnableToCompleteException {
57  		//this.logger = logger;
58  		this.context = context;
59  
60  		TypeOracle oracle = context.getTypeOracle();
61  		JClassType toGenerate = oracle.findType(typeName).isInterface();
62  		if (toGenerate == null) {
63  			logger.log(TreeLogger.ERROR, typeName + " is not an interface type");
64  			throw new UnableToCompleteException();
65  		}
66  
67  		String packageName = toGenerate.getPackage().getName();
68  		String simpleSourceName = toGenerate.getName().replace('.', '_') + "_Impl";
69  		PrintWriter pw = context.tryCreate(logger, packageName, simpleSourceName);
70  		if (pw == null) {
71  			return packageName + "." + simpleSourceName;
72  		}
73  
74  		ColumnSetModel columnSet = new ColumnSetModel(toGenerate, context, logger, names);
75  
76  		//public class X implements X {
77  		ClassSourceFileComposerFactory factory = new ClassSourceFileComposerFactory(packageName, simpleSourceName);
78  		factory.addImplementedInterface(typeName);
79  
80  		factory.addImport(Name.getSourceNameForClass(GWT.class));
81  		factory.addImport(Name.getSourceNameForClass(CellTable.class));
82  		factory.addImport(Name.getSourceNameForClass(HasDataFlushableEditor.class));
83  		factory.addImport(Name.getSourceNameForClass(Column.class));
84  		factory.addImport(Name.getSourceNameForClass(HasHorizontalAlignment.class));
85  		factory.addImport(Name.getSourceNameForClass(HasVerticalAlignment.class));
86  		factory.addImport(Name.getSourceNameForClass(FieldUpdater.class));
87  		factory.addImport(columnSet.getBeanName());
88  
89  		SourceWriter sw = factory.createSourceWriter(context, pw);
90  
91  
92  		//wire up the factory, if any
93  		if (columnSet.hasFactory()) {
94  			names.add("factory");
95  			sw.println("private %1$s factory;", columnSet.getFactoryClassName());
96  			sw.println("public void setFactory(%1$s factory) {", columnSet.getFactoryClassName());
97  			sw.indent();
98  			sw.println("assert factory != null && this.factory == null : \"Factory cannot be reset, and factory cannot be set as null\";");
99  			sw.println("this.factory = factory;");
100 			sw.outdent();
101 			sw.println("}");
102 		}
103 
104 		// generate column methods
105 		for (ColumnModel c : columnSet.getColumnModels()) {
106 			// make the field 
107 			// TODO: no sense in building multiple copies, right?
108 			sw.println("private %1$s %2$s;", c.getCellClassName(), c.getCellFieldName());
109 
110 			sw.println("private Column<%1$s,%2$s> %3$s;", columnSet.getBeanName(), c.getCellDataTypeName(), c.getColumnFieldName());
111 			sw.println();
112 
113 			// make the method: public MyCell myDataMember() {
114 			//sw.println("@Override");//jdk 5 doesnt like this
115 			sw.println("public %1$s %2$s() {", c.getCellClassName(), c.getMethodName());
116 			sw.indent();
117 			sw.println("if (%s == null) {", c.getCellFieldName());
118 			sw.indent();
119 
120 			//create the cell
121 			sw.println("%1$s = %2$s;", c.getCellFieldName(), c.getCellCreateExpression());
122 
123 			//create the column - probably should be done later in the case of using HasDataFlushableEditor
124 			sw.println("%1$s = new Column<%2$s,%3$s> (%4$s) {", c.getColumnFieldName(), columnSet.getBeanName(), c.getCellDataTypeName(), c.getCellFieldName());
125 			sw.indent();
126 
127 			sw.println("@Override");
128 			sw.println("public %1$s getValue(%2$s bean) {", c.getCellDataTypeName(), columnSet.getBeanName());
129 			sw.indent();
130 			sw.println("return %1$s;", c.getGetterInModel("bean"));
131 			sw.outdent();
132 			sw.println("}");
133 
134 			sw.outdent();// end anon Column class
135 			sw.println("};");
136 
137 			// Refactor at least this part out, in anticipation of a proper link to the Editor framework
138 			// TODO this is done by replacement right now, fix that.
139 			if (c.isEditable()) {
140 				if (!c.hasCustomFieldUpdater()) {
141 					sw.println("%1$s.setFieldUpdater(new FieldUpdater<%2$s,%3$s>() {", c.getColumnFieldName(), columnSet.getBeanName(), c.getCellDataTypeName());
142 					sw.indent();
143 
144 					sw.println("public void update(int index, %1$s object, %2$s value) {", columnSet.getBeanName(), c.getCellDataTypeName());
145 					sw.indent();
146 					sw.println("%1$s;", c.getSetterInModel("object", "value"));
147 					sw.outdent();
148 					sw.println("}");
149 
150 					sw.outdent();// end anon FieldUpdater class
151 					sw.println("});");
152 				} else {
153 					sw.println("%1$s.setFieldUpdater(GWT.<%2$s>create(%2$s.class));", c.getColumnFieldName(), c.getFieldUpdaterType().getQualifiedSourceName());
154 				}
155 			}
156 			sw.println("%1$s.setHorizontalAlignment(%2$s);", c.getColumnFieldName(), c.getHorizontalAlignment());
157 			sw.println("%1$s.setVerticalAlignment(%2$s);", c.getColumnFieldName(), c.getVerticalAlignment());
158 
159 			if (supportsSortable()) {
160 				sw.println("%1$s.setSortable(%2$s);", c.getColumnFieldName(), c.isSortable());
161 			} else {
162 				if (c.isSortable()) {
163 					logger.log(Type.WARN, "Your version of GWT does not appear to support Column.setSortable, compilation may fail.");
164 				}
165 			}
166 			//end column creation/setup
167 
168 			sw.outdent();
169 			sw.println("}");// end column/cell creation
170 			sw.println("return %s;", c.getCellFieldName());
171 			sw.outdent();
172 			sw.println("}");
173 		}
174 
175 		// generate configure methods
176 
177 		// simple overload
178 		sw.println("public final void configure(CellTable<%1$s> table) {", columnSet.getBeanName());
179 		sw.indent();
180 		sw.println("configure(table, null);");
181 		sw.outdent();
182 		sw.println("}");
183 
184 		// actual heavy-lifting one
185 		sw.println("public final void configure(CellTable<%1$s> table, HasDataFlushableEditor<%1$s> ed) {", columnSet.getBeanName());
186 		sw.indent();
187 		if (columnSet.hasFactory()) {
188 			sw.println("assert factory != null : \"setFactory() must be called before configure() can be called.\";");
189 		}
190 		for (ColumnModel c : columnSet.getColumnModels()) {
191 			//wire up the cell and column
192 			sw.println("%1$s();", c.getMethodName());
193 
194 			if (c.isEditable() && !c.hasCustomFieldUpdater()) {
195 				// if there is an editor, replace the FieldUpdater
196 				sw.println("if (ed != null) {");
197 				sw.indent();
198 				sw.println("final FieldUpdater<%1$s, %2$s> wrapped = %3$s.getFieldUpdater();", columnSet.getBeanName() , c.getCellDataTypeName() ,c.getColumnFieldName());
199 				sw.println("%1$s.setFieldUpdater(ed.new PendingFieldUpdateChange<%2$s>(){", c.getColumnFieldName(), c.getCellDataTypeName());
200 				sw.indent();
201 				sw.println("public void commit(int index, %1$s object, %2$s value) {", columnSet.getBeanName(), c.getCellDataTypeName());
202 				sw.indent();
203 				sw.println("wrapped.update(index, object, value);");
204 				sw.outdent();
205 				sw.println("}");
206 				sw.outdent();
207 				sw.println("});");
208 				sw.outdent();
209 				sw.println("}");
210 			}
211 
212 			// attach the column
213 			sw.println("table.addColumn(%1$s, %2$s);", c.getColumnFieldName(), c.getHeaderValue());
214 		}
215 		sw.outdent();
216 		sw.println("}");
217 
218 		sw.println("public String[] getPaths() {");
219 		sw.indent();
220 		sw.println("return %1$s;", columnSet.getPaths());
221 		sw.outdent();
222 		sw.println("}");
223 
224 		sw.commit(logger);
225 
226 		return factory.getCreatedClassName();
227 	}
