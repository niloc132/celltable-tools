29  /**
30   * Simplifies creation of {@link com.google.gwt.user.cellview.client.Column} objects for a 
31   * {@link CellTable}. Interfaces extending this may be declared and requested using GWT.create, and
32   * used to add columns to the CellTable.
33   * 
34   * Declared methods should return a type which implements {@link com.google.gwt.cell.client.Cell},
35   * and may be named for the properties available in the data to be drawn by the CellTable instance.
36   * As with the {@link com.google.gwt.editor.client.Editor} framework and its implementors, these 
37   * methods may be named as desired, with a {@link com.google.gwt.editor.client.Editor.Path}
38   * annotation declaring how the data may be accessed. Again, as with Editor, the path provided may
39   * be blank, indicating the row itself, or may reference nested properties.
40   * 
41   * Cell instances will be created through the use of GWT.create, allowing for replacement or 
42   * generation. If you wish to provide your own implementation, either create the Column object
43   * manually, or see {@link ColumnsWithFactory} for a way to provide a factory instance.
44   * 
45   * Once a Cell has been created, either by calling the declared methods directly, or by calling
46   * configure(...), it will be retained, so that subsequent calls will get the same instance. At this
47   * time there is no support for getting access to the Column instances.
48   * 
49   * Further extending a subinterface of Columns to add more columns is not supported at this time.
50   * 
51   * @author Colin Alworth
52   * 
53   * @param <T> The type of data that will be rendered in the CellTable
54   *
55   */
56  public interface Columns<T> {
57  	/**
58  	 * Adds each of the Cells to the given table, wrapped in a Column with the options specified
59  	 * configured on them. Columns are added in the order they are declared.
60  	 * 
61  	 * @param cellTable
62  	 */
63  	void configure(CellTable<T> cellTable);
64  
65  	/**
66  	 * As configure(CellTable<T>), except will also bind the columns to the editor, so that when the
67  	 * editor system flushes, the data will be applied to the model.
68  	 * 
69  	 * Because data will not be applied to the model until the flush occurs, some data can be
70  	 * somewhat inconsistent - the Cells will need to track their changed data correctly until the
71  	 * data is flushed, and other cells won't be able to get the updated value until then.
72  	 * 
73  	 * TODO Consider adding a configuration option to allow immediate flushing (other than not
74  	 * passing an editor)
75  	 * 
76  	 * @param cellTable
77  	 * @param editor
78  	 */
79  	void configure(CellTable<T> cellTable, HasDataFlushableEditor<T> editor);
80  
81  
82  	/**
83  	 * As in the Editor Driver classes, returns the property names that are accessed by this
84  	 * instance. At this time, primitive properties are returned, unlike in the Driver classes, and
85  	 * nested properties may be returned in an inconsistent manner.
86  	 * @return
87  	 */
88  	String[] getPaths();
89  
90  
91  
92  	/**
93  	 * Sets the alignment of the {@link Column}, as would be done using setVerticalAlignment and
94  	 * setHorizontalAlignment.
95  	 *
96  	 */
97  	@Documented
98  	@Retention(RetentionPolicy.RUNTIME)
99  	@Target(ElementType.METHOD)
100 	public @interface Alignment {
101 		public enum HorizontalAlign { CENTER, JUSTIFY, LEFT, RIGHT, NONE }
102 		public enum VerticalAlign { TOP, MIDDLE, BOTTOM, NONE }
103 		HorizontalAlign horizontal() default HorizontalAlign.NONE;
104 		VerticalAlign vertical() default VerticalAlign.NONE;
105 	}
106 
107 	/**
108 	 * Sets a static string to be used as the header of the column to be generated. Will be used
109 	 * when the column is added to the CellTable using one of the configure() methods.
110 	 * 
111 	 * At this time, cannot be internationalized.
112 	 *
113 	 */
114 	@Documented
115 	@Retention(RetentionPolicy.RUNTIME)
116 	@Target(ElementType.METHOD)
117 	public @interface Header {
118 		String value();
119 	}
120 
121 	/**
122 	 * Marks the Column as Editable, so a {@link FieldUpdater} will be created and used to call the
123 	 * matching set method for the property this column is assigned to.
124 	 * 
125 	 * Will not work for @Path(""). Nested paths are supported.
126 	 * 
127 	 * There is a known issue in GWT (http://code.google.com/p/google-web-toolkit/issues/detail?id=5981)
128 	 * where a CellTable attached to an Editor through HasDataFlushableEditor cannot have values changed. This
129 	 * annotation does nothing to help with the issue.
130 	 * 
131 	 * 
132 	 * This can also be applied to the entire type
133 	 *
134 	 */
135 	@Documented
136 	@Retention(RetentionPolicy.RUNTIME)
137 	@Target(ElementType.METHOD)
138 	public @interface Editable {
139 		/**
140 		 * If a {@link FieldUpdater} type is specified, @Path and @ConvertedWith will be ignored
141 		 * when data is being saved - the specified type must handle all of this.
142 		 * 
143 		 * Specified instance will be created with GWT.create().
144 		 */
145 		@SuppressWarnings("rawtypes")
146 		Class<? extends FieldUpdater> value() default FieldUpdater.class;
147 	}
148 
149 	/**
150 	 * Allows a type implementing {@link DataConverter} to be provided which will accept data from
151 	 * the get method this Column is mapped to and transform it before passing it the Cell instance
152 	 * and back again. The given type will be created using GWT.create as needed and not cached.
153 	 */
154 	@Documented
155 	@Retention(RetentionPolicy.RUNTIME)
156 	@Target(ElementType.METHOD)
157 	public @interface ConvertedWith {
158 		Class<? extends DataConverter<?,?>> value();
159 	}
160 
161 	/**
162 	 * Indicates that the Column should be set as sortable for the client.
163 	 * 
164 	 * Supported by GWT 2.2 and later.
165 	 */
166 	@Documented
167 	@Retention(RetentionPolicy.RUNTIME)
168 	@Target(ElementType.METHOD)
169 	public @interface Sortable {
170 	}
